import Link from 'next/link'
import Image from 'next/image'
import { useState } from "react";

# The Team

export function Card(props) {
  const [showDescription, setShowDescription] = useState(false);
  const emailPrefix = "mailto:";

  // Determine if a single link exists
  const hasSingleLink = props.pagelink && props.pagetext;

  // If pagelinks is defined and an array, use it; otherwise, fall back to single link if it exists
  const linkList = Array.isArray(props.pagelinks)
    ? props.pagelinks
    : hasSingleLink
      ? [{ url: props.pagelink, text: props.pagetext }]
      : [];

  return (
    <div className="transition-all duration-300 mb-4 w-full md:w-1/3 grow-0 bg-blue-500/15 hover:shadow-blue-500 hover:bg-blue-500/20 hover:shadow-md m-2 rounded-lg">
      <div className="p-4">
        {/* Image */}
        <div className="w-full h-48 relative">
          <Image
            src={props.imgpath}
            alt={props.name}
            fill
            className="rounded-t-lg object-cover"
          />
        </div>

        {/* Name */}
        <p className="mt-2 font-bold">{props.name}</p>

        {/* Email */}
        <Link href={emailPrefix + props.email}>
          <p className="text-blue-700 hover:text-blue-400">Email</p>
        </Link>

        {/* Links */}
        {linkList.length > 0 && (
          <div className="mt-1 space-y-1">
            {linkList.map((link, index) => (
              <a
                key={index}
                href={link.url}
                target="_blank"
                rel="noopener noreferrer"
                className="block text-blue-700 hover:text-blue-400"
              >
                {link.text}
              </a>
            ))}
          </div>
        )}

        {/* Address */}
        <p className="text-sm italic mt-1">{props.address}</p>

        {/* Description toggle */}
        <button
          onClick={() => setShowDescription(!showDescription)}
          className="mt-2 text-sm text-blue-700 hover:text-blue-400"
        >
          {showDescription ? "Hide Description ▲" : "Show Description ▼"}
        </button>

        {showDescription && <p className="mt-2">{props.description}</p>}
      </div>
    </div>
  );
}
